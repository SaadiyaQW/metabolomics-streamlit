虚拟环境搭建：
# 创建名为 metabolomics 的环境，指定 Python 3.10
conda create -n metabolomics python=3.10 -y

# 激活环境
conda activate metabolomics

conda install pandas numpy scipy -y

conda install scikit-learn statsmodels -y

conda install matplotlib seaborn plotly -y

pip install streamlit

pip install ropls          # OPLS-DA分析

pip install bioservices    # 调用KEGG等数据库

conda install openpyxl xlrd -y        # Excel文件
pip install python-pptx               # PowerPoint（如需生成报告）

pip install gseapy        # GSEA富集分析

pip install shap          # 模型可解释性
pip install lime          # 局部解释模型
pip install umap-learn    # UMAP降维（替代t-SNE）
pip install hdbscan       # 聚类分析
